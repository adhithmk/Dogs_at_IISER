Auto Image URL Extractor for Dogs of IISER

Important: This tool helps you automatically extract image URLs from your Google Drive folder. You'll need to run this in your browser while logged into your Google account.

Step 2: Run the Auto Extractor

Open your browser's developer console (F12 or Ctrl+Shift+I) and paste this code:

async function extractAllDogImages() { const dogs = []; const folderNames = []; // Get all dog folders const folderElements = document.querySelectorAll('[data-id]'); for (let element of folderElements) { const text = element.textContent || element.innerText; if (text.includes('Dog ') && /^\s*Dog \d+\s*$/.test(text.trim())) { folderNames.push(text.trim()); } } console.log('Found dog folders:', folderNames); for (let folderName of folderNames) { console.log(`Processing ${folderName}...`); // Click on the folder const folderLink = Array.from(document.querySelectorAll('a')).find(a => a.textContent.includes(folderName) ); if (folderLink) { folderLink.click(); await new Promise(resolve => setTimeout(resolve, 2000)); // Wait for folder to load // Get all image links in this folder const imageLinks = []; const links = document.querySelectorAll('a[href*="/file/d/"]'); for (let link of links) { const href = link.href; if (href.includes('/file/d/')) { const fileId = href.match(/\/file\/d\/([^/?]+)/); if (fileId) { const directUrl = `https://drive.google.com/uc?export=view&id=${fileId[1]}`; imageLinks.push(directUrl); } } } dogs.push({ name: folderName, images: imageLinks }); console.log(`${folderName}: Found ${imageLinks.length} images`); // Go back to main folder const backButton = document.querySelector('[aria-label="Back"]'); if (backButton) backButton.click(); await new Promise(resolve => setTimeout(resolve, 2000)); } } // Generate CSV let csv = 'Aadhaar ID name,Pictures\n'; dogs.forEach(dog => { const imageUrls = dog.images.join(','); csv += `${dog.name},"${imageUrls}"\n`; }); console.log('Generated CSV:'); console.log(csv); // Copy to clipboard navigator.clipboard.writeText(csv).then(() => { console.log('CSV copied to clipboard!'); alert('CSV data copied to clipboard! Check console for full data.'); }); return dogs; } // Run the function extractAllDogImages();

Step 3: Update Your CSV

After running the code, the CSV data will be copied to your clipboard. Paste it into your CSV file.
